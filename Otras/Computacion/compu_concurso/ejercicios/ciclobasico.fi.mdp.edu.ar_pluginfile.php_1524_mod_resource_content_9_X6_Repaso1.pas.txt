program Funcion_x_Tramos;
var
    x,f,prom:real;
    contsi,contno:byte;
begin
contsi:=0; contno:=0; prom:=0;
readln(x);
while x<>1000 do
      begin
      if (x>=-5)and(x<=5)then
            begin
            contsi:=contsi+1;
            prom:=prom+x;
            if x<-1 then
                      f:=abs(x+1) {o bien f:=-(x+1)}
           else
                      if x<1 then
                              f:=sqr(x)-1
                     else
                              f:=ln(x);
           writeln(f:5:2);
           end
     else
           begin
           contno:=contno+1;
           writeln(x:4:2,' no pertenece al dominio');
           end;
     readln(x);
     end;
 writeln(contno,' valores ingresados no pertenecen al dominio');
 if contsi<>0 then
      writeln('el promedio de valores de x pertenecientes al dominio es:',prom/contsi:5:2)
 else
     writeln('no hay valores de x pertenecientes al dominio');
end.
